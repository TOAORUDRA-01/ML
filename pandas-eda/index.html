Python
import pandas as pd
import numpy as np

# Create from list (auto integer index)
s1 = pd.Series([10, 20, 30, 40])

# Create with custom index
salaries = pd.Series(
    [92000, 87000, 71000, 65000],
    index=['mumbai', 'delhi', 'pune', 'chennai'],
    name='avg_salary'
)

# Create from dict
salaries = pd.Series({'mumbai': 92000, 'delhi': 87000})

# Key attributes
salaries.values      # ndarray: [92000 87000 ...]
salaries.index       # Index: ['mumbai', 'delhi', ...]
salaries.dtype       # dtype('float64')
salaries.shape       # (4,)
salaries.name        # 'avg_salary'

# Vectorised operations — just like NumPy
salaries * 1.1           # 10% raise for all cities
salaries[salaries > 80000] # boolean mask
salaries.sort_values()   # ascending by default
salaries.rank()          # rank 1=smallest

# Access by label vs position
salaries['mumbai']      # label access: 92000
salaries.loc['mumbai']  # same, explicit
salaries.iloc[0]        # position access: 92000

# String methods via .str accessor
cities = pd.Series(['Mumbai', 'New Delhi', 'Pune'])
cities.str.lower()
cities.str.contains('Delhi')
cities.str.replace(' ', '_')

Python — Missing Values
# Detect
df.isnull()              # bool DataFrame — True where NaN
df.isnull().sum()        # count nulls per column
df.isnull().sum() / len(df)  # proportion of nulls
df.isnull().any(axis=1)  # rows with ANY null

# Strategy 1: Drop
df.dropna()                           # drop rows with ANY null
df.dropna(subset=['salary'])          # only if salary is null
df.dropna(thresh=5)                   # keep rows with ≥5 non-null
df.dropna(axis=1, thresh=int(0.7*len(df))) # drop cols with >30% nulls

# Strategy 2: Fill — numeric
df['salary'].fillna(df['salary'].mean())   # mean imputation
df['salary'].fillna(df['salary'].median()) # median (robust to outliers)
df['salary'].fillna(method='ffill')        # forward fill (time series)
df['salary'].fillna(method='bfill')        # backward fill

# Strategy 3: Fill — categorical
df['city'].fillna(df['city'].mode()[0])    # most frequent
df['city'].fillna('Unknown')               # explicit sentinel

# Strategy 4: Interpolate (time series)
df['price'].interpolate(method='linear')
df['price'].interpolate(method='time')

Duplicates & Data Types

Python — Cleaning continued
# Duplicates
df.duplicated().sum()          # count duplicates
df.duplicated(subset=['name','age'])  # on specific columns
df.drop_duplicates(keep='first')    # keep first occurrence
df.drop_duplicates(inplace=True)    # in-place

# Data type correction
df['age'] = df['age'].astype(int)
df['salary'] = pd.to_numeric(df['salary'], errors='coerce')
df['date'] = pd.to_datetime(df['date'], format='%Y-%m-%d')
df['city'] = df['city'].astype('category')  # saves memory!

# String cleaning pipeline
df['name'] = (df['name']
    .str.strip()          # remove leading/trailing whitespace
    .str.lower()          # normalise case
    .str.replace(r'\s+', '_', regex=True)  # spaces → underscore
)

# Outlier detection — IQR method
Q1  = df['salary'].quantile(0.25)
Q3  = df['salary'].quantile(0.75)
IQR = Q3 - Q1
lower, upper = Q1 - 1.5*IQR, Q3 + 1.5*IQR
outliers = df[(df['salary'] < lower) | (df['salary'] > upper)]
df_clean = df.clip(lower, upper, subset=['salary'])  # cap instead of drop
★ The three strategies for missing data

Drop when <5% missing and randomly distributed (MCAR).   Impute with mean/median for numeric when 5–30% missing.   Flag + fill for >30% missing — add a binary indicator column salary_was_null to preserve the signal that the value was absent (often predictive!).

Section 04

Transformations & apply()

Pandas gives you vectorised operations, apply() for custom logic, and map() for element-wise substitution. Know when to use each.

apply() — the Swiss Army Knife

Python — apply / map / applymap
# apply to a Series — element-wise
df['salary'].apply(lambda x: 'high' if x > 80000 else 'low')

# apply to a DataFrame — axis=0 (cols) or axis=1 (rows)
df.apply(np.mean, axis=0)    # mean of each column
df.apply(np.sum,  axis=1)    # row sums

# Custom multi-column logic
def risk_level(row):
    if row['churn'] == 1 and row['salary'] < 60000:
        return 'high'
    elif row['churn'] == 1:
        return 'medium'
    return 'low'

df['risk'] = df.apply(risk_level, axis=1)

# map — replace values via dict (Series only)
df['churn_label'] = df['churn'].map({0: 'retained', 1: 'churned'})

# pd.cut — bin numeric → categorical
df['age_group'] = pd.cut(
    df['age'],
    bins=[0, 25, 35, 50, 100],
    labels=['youth', 'adult', 'senior', 'elder']
)

# pd.qcut — equal-frequency bins (quantile-based)
df['salary_quartile'] = pd.qcut(df['salary'], q=4,
                              labels=['Q1','Q2','Q3','Q4'])

# np.where — vectorised if/else
df['high_value'] = np.where(df['salary'] > 75000, 1, 0)
✓ Performance tip

Prefer vectorised operations (column arithmetic, .str, np.where) over .apply() — they run 10–100× faster because they delegate to C. Use .apply() only when the logic genuinely cannot be vectorised.

Section 05

Indexing & Selection

Mastering loc vs iloc is the difference between clean code and indexing bugs. Know the four access methods cold.

MethodSelects byExampleReturns
df[col]Column namedf['age']Series
df.loc[r, c]Labelsdf.loc[0:3, 'age':'city']DataFrame
df.iloc[r, c]Integer positionsdf.iloc[0:3, 1:4]DataFrame
df[mask]Boolean Seriesdf[df.age > 30]DataFrame
Python — loc / iloc / boolean
# loc — label-based (slice is INCLUSIVE on both ends)
df.loc[2:5, 'age':'salary']   # rows 2-5, cols age to salary
df.loc[df.city == 'Mumbai', ['name','salary']]

# iloc — position-based (slice EXCLUDES right end like Python)
df.iloc[0:5, 1:4]            # rows 0-4, columns 1-3
df.iloc[::2]                  # every other row
df.iloc[-5:]                  # last 5 rows

# Boolean indexing — the most important pattern
high_earners = df[df['salary'] > 80000]

# Multiple conditions — use & | ~ (not and/or/not!)
mask = (df['age'] > 30) & (df['city'] == 'Mumbai') & (df['churn'] == 0)
df[mask]

# .query() — cleaner syntax for boolean selection
df.query("age > 30 and city == 'Mumbai' and churn == 0")

# isin — check membership
df[df['city'].isin(['Mumbai', 'Delhi', 'Pune'])]

# between — range check
df[df['age'].between(25, 40)]

# Setting values with loc (avoid SettingWithCopyWarning)
df.loc[df['salary'].isnull(), 'salary'] = df['salary'].median()

// Interactive: Group Salary by City
Python — GroupBy
# Basic groupby
df.groupby('city')['salary'].mean()

# Multiple aggregations with agg()
summary = df.groupby('city').agg(
    avg_salary = ('salary', 'mean'),
    med_salary = ('salary', 'median'),
    n_customers= ('name',   'count'),
    churn_rate = ('churn',  'mean'),    # mean of 0/1 = rate
    max_age    = ('age',    'max')
).reset_index()

# Group by multiple columns
df.groupby(['city', 'churn'])[['salary','age']].mean()

# transform — broadcast result back to original shape
# Great for "salary vs city average" features
df['city_avg_sal'] = df.groupby('city')['salary'].transform('mean')
df['sal_vs_avg']    = df['salary'] - df['city_avg_sal']

# Custom aggregation functions
cv = lambda x: x.std() / x.mean() # coefficient of variation
df.groupby('city')['salary'].agg(['mean', 'std', cv])

# filter — keep only groups satisfying condition
big_cities = df.groupby('city').filter(lambda g: len(g) >= 50)

# pivot_table — GroupBy with spreadsheet syntax
pivot = df.pivot_table(
    values='salary', index='city',
    columns='churn', aggfunc='mean', fill_value=0
)
Section 07

Merge, Join & Concat

Combining DataFrames is the bread and butter of real data work — you almost never receive data in one clean table.

Join Types — Visual

INNER JOIN

Only matching rows from both tables.

LEFT JOIN

All left rows + matching right (NaN if no match).

OUTER JOIN

All rows from both sides (NaN where no match).

Python — Merge / Join / Concat
# pd.merge — SQL-style join
customers = pd.DataFrame({'id':[1,2,3], 'name':['A','B','C']})
orders    = pd.DataFrame({'cust_id':[1,2,2,4], 'amount':[100,200,150,80]})

# Inner join — only customers who ordered
pd.merge(customers, orders,
        left_on='id', right_on='cust_id', how='inner')

# Left join — all customers, NaN if no orders
pd.merge(customers, orders,
        left_on='id', right_on='cust_id', how='left')

# Merge on index
df1.merge(df2, left_index=True, right_index=True, how='inner')

# pd.concat — stack DataFrames
# Vertically (more rows):
combined = pd.concat([df_2022, df_2023, df_2024], ignore_index=True)

# Horizontally (more columns):
features = pd.concat([df_basic, df_derived], axis=1)

# Detect merge problems
merged = pd.merge(df1, df2, on='id', how='outer', indicator=True)
merged['_merge'].value_counts()  # left_only, right_only, both

Python — EDA Workflow
# ─── Step 1: Overview ───
df.shape                       # (n_rows, n_cols)
df.dtypes.value_counts()        # count int64/float64/object
df.describe(percentiles=[.01,.25,.5,.75,.99])

# ─── Step 2: Target ───
df['churn'].value_counts()           # class counts
df['churn'].value_counts(normalize=True) # class proportions

# ─── Step 3: Numeric features ───
num_cols = df.select_dtypes(include=['number']).columns
df[num_cols].skew()               # skewness (>1 = right skewed)
df[num_cols].kurtosis()           # heavy tails

# ─── Step 4: Categorical features ───
cat_cols = df.select_dtypes(include=['object','category']).columns
for col in cat_cols:
    print(df[col].value_counts().head(10))

# ─── Step 5: Correlations ───
corr_matrix = df[num_cols].corr()
# Target correlation
target_corr = corr_matrix['churn'].abs().sort_values(ascending=False)
print(target_corr)

# ─── Step 6: Bivariate — feature vs target ───
df.groupby('churn')['salary'].describe()
df.groupby('churn')[num_cols].mean().T  # transposed for readability

# ─── Step 7: Document findings ───
findings = {
    'class_imbalance': df.churn.mean(),
    'missing_cols': df.isnull().any().sum(),
    'high_corr_features': target_corr[target_corr > 0.3].index.tolist()
}

Full EDA Workflow

A complete, battle-tested EDA template you can copy-paste on any new dataset.

Python — Complete EDA Template
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from scipy import stats

# ════════════════════════════════════════
# PHASE 1: LOAD & QUICK OVERVIEW
# ════════════════════════════════════════
df = pd.read_csv('data.csv')
print(f"Shape: {df.shape}")
print(f"Memory: {df.memory_usage(deep=True).sum()/1e6:.1f} MB")
df.head()
df.info()
df.describe(percentiles=[.01,.05,.25,.5,.75,.95,.99])

# ════════════════════════════════════════
# PHASE 2: DATA QUALITY REPORT
# ════════════════════════════════════════
quality = pd.DataFrame({
    'dtype':    df.dtypes,
    'nulls':    df.isnull().sum(),
    'null_pct': df.isnull().mean().round(3),
    'unique':   df.nunique(),
    'zeros':    (df == 0).sum()
})
print(quality.sort_values('null_pct', ascending=False))

# ════════════════════════════════════════
# PHASE 3: TARGET ANALYSIS
# ════════════════════════════════════════
TARGET = 'churn'
print(df[TARGET].value_counts())
print(df[TARGET].value_counts(normalize=True))

# ════════════════════════════════════════
# PHASE 4: NUMERIC FEATURES
# ════════════════════════════════════════
num_cols = df.select_dtypes('number').columns.drop(TARGET, errors='ignore')
fig, axes = plt.subplots(len(num_cols), 3, figsize=(15, 4*len(num_cols)))

for i, col in enumerate(num_cols):
    # Distribution
    df[col].hist(bins=30, ax=axes[i,0]); axes[i,0].set_title(col)
    # vs Target
    df.boxplot(col, by=TARGET, ax=axes[i,1])
    # QQ-plot (normality check)
    stats.probplot(df[col].dropna(), plot=axes[i,2])

# ════════════════════════════════════════
# PHASE 5: CATEGORICAL FEATURES
# ════════════════════════════════════════
cat_cols = df.select_dtypes(include=['object','category']).columns
for col in cat_cols:
    churn_rate = df.groupby(col)[TARGET].mean().sort_values()
    print(f"\n{col} churn rates:")
    print(churn_rate)

# ════════════════════════════════════════
# PHASE 6: CORRELATIONS
# ════════════════════════════════════════
fig, ax = plt.subplots(figsize=(10, 8))
corr = df[num_cols.tolist() + [TARGET]].corr()
mask = np.triu(np.ones_like(corr, dtype=bool))  # upper triangle
sns.heatmap(corr, mask=mask, annot=True, fmt='.2f',
           cmap='RdYlGn', center=0, square=True, ax=ax)

plt.tight_layout()
plt.savefig('full_eda.png', dpi=150, bbox_inches='tight')

★ EDA Golden Rules

1. Always look at raw data first — df.sample(20) before any code.
2. Question every number — if mean salary is ₹500K, is that realistic?
3. Look at the target separately per group — groupby(cat).target.mean() reveals patterns instantly.
4. Log-transform right-skewed features — salary, price, counts are almost always skewed.
5. Document findings as you go — write hypotheses as comments before you build models.
